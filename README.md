```
{
  "username": "kengerlwl", # 用户名
  "repository" : "MDimg", # 仓库名
  "proxy": true, # 针对特殊图片，是否使用代理
  "wordpress": { # 上传到wordpress的配置
    "host": "http://host:8081",
    "username": "***",
    "password": "***"
  }
}
```
